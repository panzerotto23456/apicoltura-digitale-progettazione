#include "esp_camera.h"
#include <WiFi.h>
#include <HTTPClient.h>
#include <base64.h>

// --- CONFIGURAZIONE WIFI E API ---
const char* ssid = "IL_TUO_WIFI";
const char* password = "LA_TUA_PASSWORD";
const char* restdb_url = "https://TUO_DATABASE.restdb.io/rest/NOME_TABELLA";
const char* restdb_api_key = "LA_TUA_API_KEY_QUI"; // "x-apikey"

// --- PINOUT ESP32-CAM (AI-THINKER) ---
#define PWDN_GPIO_NUM     32
#define RESET_GPIO_NUM    -1
#define XCLK_GPIO_NUM      0
#define SIOD_GPIO_NUM     26
#define SIOC_GPIO_NUM     27
#define Y9_GPIO_NUM       35
#define Y8_GPIO_NUM       34
#define Y7_GPIO_NUM       39
#define Y6_GPIO_NUM       36
#define Y5_GPIO_NUM       21
#define Y4_GPIO_NUM       19
#define Y3_GPIO_NUM       18
#define Y2_GPIO_NUM        5
#define VSYNC_GPIO_NUM    25
#define HREF_GPIO_NUM     23
#define PCLK_GPIO_NUM     22

// Variabili di controllo
unsigned long ultimoScatto = 0;
const long intervalloFoto = 900000; // 15 minuti in millisecondi
String ultimaFotoBase64 = "";      // Variabile per immagazzinare il "blob" per usi futuri

void setup() {
  Serial.begin(115200);
  
  // Connessione WiFi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi Connesso!");

  // Configurazione Camera
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG;
  config.frame_size = FRAMESIZE_QVGA; 
  config.jpeg_quality = 12;
  config.fb_count = 1;

  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("Errore Camera: 0x%x", err);
    return;
  }
  Serial.println("Sistema pronto!");
}

void scattaEInvia() {
  camera_fb_t * fb = esp_camera_fb_get();
  if (!fb) {
    Serial.println("Errore acquisizione!");
    return;
  }

  // 1. Convertiamo il buffer (blob) in Base64 per restdb.io
  // Lo salviamo nella variabile globale per uso futuro
  ultimaFotoBase64 = base64::encode(fb->buf, fb->len);
  
  Serial.printf("Foto catturata! Dimensione: %zu bytes\n", fb->len);

  // 2. Invio al Database REST
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(restdb_url);
    
    // Header obbligatori per restdb.io
    http.addHeader("Content-Type", "application/json");
    http.addHeader("x-apikey", restdb_api_key);
    http.addHeader("Cache-Control", "no-cache");

    // Costruiamo il JSON. Il campo 'mio_blob' deve esistere nella collezione
    String jsonPayload = "{\"mio_blob\":\"" + ultimaFotoBase64 + "\", \"timestamp\":\"" + String(millis()) + "\"}";

    int httpResponseCode = http.POST(jsonPayload);

    if (httpResponseCode > 0) {
      Serial.print("Inviato! Risposta server: ");
      Serial.println(httpResponseCode);
    } else {
      Serial.print("Errore nell'invio: ");
      Serial.println(http.errorToString(httpResponseCode).c_str());
    }
    http.end();
  }

  esp_camera_fb_return(fb);
}

void loop() {
  unsigned long adesso = millis();

  if (adesso - ultimoScatto >= intervalloFoto) {
    ultimoScatto = adesso;
    scattaEInvia();
  }

  if (Serial.available()) {
    String comando = Serial.readStringUntil('\n');
    comando.trim();
    if (comando == "foto") {
      scattaEInvia();
    }
  }
}
